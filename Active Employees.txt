select Cast(E.ID_NUM as varchar(18)) as ID_NUM,	
	   CASE WHEN E.FULLPART_STS = 'F'
		THEN 'FULL TIME'
		WHEN E.FULLPART_STS = 'P'
		THEN 'PART TIME'
		ELSE ''
		END AS FULL_TIME_STATUS,
	   CASE WHEN N.PREFIX = '1'
		THEN 'Dr'
		WHEN N.PREFIX = '2'
		THEN 'Mr'
		WHEN N.PREFIX = '3'
		THEN 'Mrs'
		WHEN N.PREFIX = '5'
		THEN 'Ms'
		WHEN N.PREFIX = '4'
		THEN 'Ms'
		WHEN N.PREFIX = '6'
		THEN 'Sr'
		WHEN N.PREFIX IS NULL AND B.GENDER = 'M'
		THEN 'Mr'
		WHEN N.PREFIX IS NULL AND B.GENDER = 'F'
		THEN 'Ms'
		END AS PREFIX,
	   N.FIRST_NAME,
	   N.PREFERRED_NAME,
	   N.LAST_NAME,
	   case when NU.TITLE is null 
			then N.TITLE 
			else NU.TITLE
		end as TITLE,
	   M.ATTENTION_LINE AS MALONE_USER,
	   M.ADDR_LINE_1 AS EMAIL,
	   A.ADDR_LINE_1 AS MAILING_STREET,
	   A.CITY AS MAILING_CITY,
	   A.STATE AS MAILING_STATE,
	   A.ZIP AS MAILING_ZIP,
	   P.PHONE_NUMBER AS WORK_PHONE,
	   CASE WHEN N.MOBILE_PHONE_PRIVA = 'Y'
	   THEN NULL
	   ELSE N.MOBILE_PHONE
	   END AS MOBILE_PHONE,	  
	   CASE WHEN B.GENDER = 'M'
	    THEN 'Male'
		WHEN B.GENDER = 'F'
		THEN 'Female'
		ELSE 'Unknown'
		END AS GENDER,
	   D.TABLE_DESC AS MARITAL_STS,
	   '01261000000X7J0AAK' as RecordTypeID,
	   'EX' as SYSTEM_OF_RECORD,
	   '--None--' as STUDENT_STAGE,
	  '00161000008iw63AAA' AS ACCOUNT_ID,
	  '00561000003FgayAAC' AS OWNER_ID,
	  b.CURRENT_EMPLOY_TYP,
	  CASE WHEN b.EMPLOYEE_OF_COLLEG = 'Y'
		THEN '1'
		ELSE '0'
		END AS EMPLOYEE_OF_COLLEG,
	  CASE WHEN F.ACTIVE = 'Y' and b.EMPLOYEE_OF_COLLEG = 'Y'
		THEN '1'
		ELSE '0'
	  END AS FACULTY_FLAG,
	  CASE WHEN S.EMPLOYEE_OF_COLLEG = 'Y' and b.EMPLOYEE_OF_COLLEG = 'Y'
		THEN '1'
		ELSE '0'
	  END AS STAFF_FLAG,
	  CASE WHEN ST.ID_NUM IS NOT NULL
		THEN '1'
		ELSE '0'
	  END AS STUDENT_FLAG,
	  ai.IDENTIFIER

from EMPL_MAST e
	JOIN NAME_MASTER N ON E.ID_NUM = N.ID_NUM
	left outer join NAME_MASTER_UDF NU on e.ID_NUM = NU.ID_NUM
	JOIN ADDRESS_MASTER A ON E.ID_NUM = A.ID_NUM
		AND A.ADDR_CDE = '*LHP'
	JOIN ADDRESS_MASTER M ON E.ID_NUM = M.ID_NUM
		AND M.ADDR_CDE = '*EML'
	JOIN BIOGRAPH_MASTER B ON E.ID_NUM = B.ID_NUM
	LEFT OUTER JOIN TABLE_DETAIL D ON B.MARITAL_STS = D.TABLE_VALUE
		AND D.COLUMN_NAME = 'marital_sts'
	LEFT OUTER JOIN EMPL_WRK_LOC P ON E.ID_NUM = P.ID_NUM
	LEFT OUTER JOIN FACULTY_MASTER F ON E.ID_NUM = F.ID_NUM
	LEFT OUTER JOIN z_mlc_email_group_staff S ON M.ADDR_LINE_1 = S.ADDR_LINE_1
	LEFT OUTER JOIN STUDENT_MASTER ST ON E.ID_NUM = ST.ID_NUM
	left outer join ALTERNATE_IDENTIFIER ai on e.ID_NUM = ai.ID_NUM and ai.IDENTIFIER_TYPE = 'JRM'
where --ACT_INACT_STS = 'A'
	 (TERMINATION_DTE IS NULL OR TERMINATION_DTE < REHIRE_DTE)
	and FIRST_NAME not like ('Reviewer%')
	and e.ID_NUM not in (438266)
	AND b.CURRENT_EMPLOY_TYP not in ('X', 'V')
	--AND B.EMPLOYEE_OF_COLLEG = 'Y'
